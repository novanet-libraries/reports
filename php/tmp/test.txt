SELECT
  RTRIM(Z76_BUDGET_NUMBER) BUDGET_NUMBER,
  Z76_NAME,
  CASE Z601_CREDIT_DEBIT
    WHEN 'C' THEN Z601_LOCAL_SUM / -100
    ELSE Z601_LOCAL_SUM / 100
  END LOCALSUM,
  RTRIM(Z68_ORDER_NUMBER) ORDER_NUMBER,
  Z68_ORDERING_UNIT,
  RTRIM(Z68_VENDOR_CODE) VENDOR_CODE,
  Z68_ORDER_TYPE,
  Z68_MATERIAL_TYPE,
  Z68_NO_UNITS UNITS_ORDERED,
  Z75_I_NO_UNITS UNITS_INVOICED,
  UNITS_ARRIVED,
  RTRIM(Z75_I_OBJECT_CODE) OBJECT_CODE,
  Z77_P_DATE,
  SUBSTR(Z68_REC_KEY,1,9) ADM_NUMBER,
  BIB_NUMBER,
  AUTHOR,
  TITLE,
  IMPRINT,
  ISN,
  CALLNUMBER
FROM NOV50.Z68
JOIN WEBREPORT.INVOICE_INFO ON (Z75_REC_KEY = Z68_REC_KEY)
JOIN WEBREPORT.ARRIVAL_COUNT_PER_ORDER ON (REC_KEY = Z68_REC_KEY)
JOIN NOV50.Z601 ON (Z601_REC_KEY_2 = Z75_REC_KEY_2)
JOIN NOV50.Z76 ON SUBSTR(Z601_REC_KEY, 1, 50) = Z76_BUDGET_NUMBER
JOIN WEBREPORT.ADM_BIB_LOOKUP ON SUBSTR(Z68_REC_KEY,1,9) = ADM_NUMBER
NATURAL JOIN WEBREPORT.SHORT_RECORDS
NATURAL JOIN WEBREPORT.ITEM_RECORDS
WHERE RTRIM(Z68_ORDERING_UNIT) = :ORDERUNIT
AND RTRIM(Z76_BUDGET_NUMBER) = :BUDGETNUMBER
